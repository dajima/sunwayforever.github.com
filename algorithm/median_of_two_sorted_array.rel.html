1 Algorithm: Median Of Two Sorted Array

#!/usr/bin/env python3
# -*- coding: utf-8 -*-


class Solution(object):

    def medianOfTwoSortedArray(self, A1, A2, f1, t1, f2, t2):
        """
        1 2 3 4 5
        f1  m1  t1

        6 7 8 9 10
        f2  m2  t2
        """
        m1 = (t1 - f1) // 2 + f1
        m2 = (t2 - f2) // 2 + f2

        if (A1[m1] == A2[m2]):
            return A1[m1]

        if (m1 == f1):
            return min(A1[t1], A2[t2])

        if (A1[m1] > A2[m2]):
            return self.medianOfTwoSortedArray(A1, A2, f1, m1, t2 - (m1 - f1), t2)
        else:
            return self.medianOfTwoSortedArray(A1, A2, m1, t1, f2, f2 + (t1 - m1))

A1 = [1, 2]
A2 = [0, 1]
print(Solution().medianOfTwoSortedArray(
    A1, A2, 0, len(A1) - 1, 0, len(A2) - 1))
