1 Algorithm: Quick Min Select

#!/usr/bin/env python3
# -*- coding: utf-8 -*-


class Solution(object):

    def quickMinSelect(self, A, f, t, n):
        """
        1 2 3 4 5 6
        0   2     5
        f   s     t
        """
        if (n <= 0 or f >= t):
            return
        if (n > t - f + 1):
            n = t - f + 1
        s = self.partition(A, f, t)
        self.quickMinSelect(A, f, s - 1, n)
        self.quickMinSelect(A, s + 1, t, n - s + f - 1)

    def partition(self, A, f, t):
        """
        1 2 3 4 5
        s f     t
        """
        s = f
        f += 1
        for i in range(f, t + 1):
            if (A[f] < A[s]):
                A[f], A[s + 1] = A[s + 1], A[f]
                A[s], A[s + 1] = A[s + 1], A[s]
                s += 1
            f += 1
        return s

A = [2, 6, 1, 2, 6, 4, 5, 6, 10, -1, 100, -1, 9, 8]
Solution().quickMinSelect(A, 0, len(A) - 1, 5)
print(A)
