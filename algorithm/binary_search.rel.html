1 Algorithm: Binary Search

# -*- coding: utf-8 -*-
class Solution(object):

    def binarySearch(self, A, value):
        """
        1 2 4 5 : 2 => 1
        returns: i for A[i] == value
        """
        f = 0
        t = len(A) - 1

        """
        循环不变量:
        A[f..t] 之外的部分不满足 A[x] == value
        """
        while (True):
            if (f > t):
                return -1
            else:
                mid = int((f + t) / 2)
                if (A[mid] == value):
                    return mid
                elif (A[mid] > value):
                    t = mid - 1
                else:
                    f = mid + 1

    def binarySearchLarger(self, A, value):
        """
        1 2 4 5 : 3 => 2
        returns: min(i for  A[i] >= value)
        """
        f = 0
        t = len(A) - 1
        ret = -1
        while (True):
            if (f > t):
                return ret
            else:
                mid = int((f + t) / 2)
                if (A[mid] == value):
                    return mid
                elif (A[mid] > value):
                    t = mid - 1
                    ret = mid
                else:
                    f = mid + 1

print(Solution().binarySearch([1, 2, 4, 5], 1))
print(Solution().binarySearch([], 1))
print(Solution().binarySearchLarger([1, 2, 4, 5], 3))
