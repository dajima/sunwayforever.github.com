# -*- coding: utf-8 -*-
import math


class Solution(object):

    def PARENT(self, i):
        return int((i - 1) / 2)

    def LEFT(self, i):
        return i * 2 + 1

    def RIGHT(self, i):
        return i * 2 + 2

    def maxHeapify(self, A, i, size):
        """
        """
        left = self.LEFT(i)
        right = self.RIGHT(i)

        if (left >= size):
            leftValue = -math.inf
        else:
            leftValue = A[left]

        if (right >= size):
            rightValue = -math.inf
        else:
            rightValue = A[right]

        if (A[i] >= max(leftValue, rightValue)):
            return

        if (leftValue >= rightValue):
            A[i], A[left] = A[left], A[i]
            self.maxHeapify(A, left, size)
        else:
            A[i], A[right] = A[right], A[i]
            self.maxHeapify(A, right, size)

    def buildMaxHeap(self, A):
        """
        """
        start = self.PARENT(len(A) - 1)
        """
        循环不变量:
        LEFT[i] 和 RIGHT[i] 是最大堆
        """
        for i in range(start, -1, -1):
            self.maxHeapify(A, i, len(A))

    def heapSort(self, A):
        """
        """
        self.buildMaxHeap(A)
        size = len(A)
        while (size > 1):
            A[0], A[size - 1] = A[size - 1], A[0]
            size -= 1
            self.maxHeapify(A, 0, size)

    def __climbUp(self, A, i):
        """
        """
        while (i != 0):
            j = self.PARENT(i)
            if (A[i] <= A[j]):
                break
            A[i], A[j] = A[j], A[i]
            i = j

    def assertHeapify(self, A):
        """
        """
        size = self.PARENT(len(A) - 1) + 1
        for i in range(0, size):
            left = self.LEFT(i)
            right = self.RIGHT(i)

            if (left >= size):
                leftValue = -math.inf
            else:
                leftValue = A[left]

            if (right >= size):
                rightValue = -math.inf
            else:
                rightValue = A[right]

            assert(A[i] >= leftValue and A[i] >= rightValue)

    def heapInsert(self, A, n):
        """
        """
        A.append(n)

        i = len(A) - 1
        self.__climbUp(A, i)

    def heapRemove(self, A, i):
        """
        """
        A[i] = A[-1]
        del(A[-1])
        if (i == len(A)):
            return
        self.maxHeapify(A, i, len(A))
        self.__climbUp(A, i)
