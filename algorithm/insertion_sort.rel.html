1 Algorithm: Insertion Sort

class Solution(object):
    def insertionSort(self, A):
        """
        1 2 4 3
          j   i
        """
        for i in range(1, len(A)):
            for j in range(i-1, -1, -1):
                if (A[j] < A[j+1]):
                    break;
                A[j], A[j+1] = A[j+1], A[j]
        print(A)

    def insertionSort2(self, A):
        """
        1 2 4 3
            j i
        """
        for i in range(1, len(A)):
            current = A[i]
            for j in range(0, i):
                if (A[j] > current):
                    break
            else:
                j += 1

            for k in range(i, j, -1):
                A[k] = A[k-1]
            A[j] = current

        print(A)
