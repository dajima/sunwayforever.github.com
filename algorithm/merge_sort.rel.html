# -*- coding: utf-8 -*-  
import math


class Solution(object):

    def mergeSort(self, A, f, t):
        """
        """
        if (f >= t):
            return

        mid = int((f + t) / 2)
        self.mergeSort(A, f, mid)
        self.mergeSort(A, mid + 1, t)
        self.mergeInPlace(A, f, mid, t)

    def mergeInPlace(self, A, p, q, r):
        """
        1 2 5 2 3 6 
        p   q     r

        循环不变量:
        A[0..(p-1)] 是升序排列的且都不大于 A[p]
        """
        # test hello
        while (True):
            # array_2 is done
            if (q + 1 > r):
                break
            # array_1 is done
            if (p > q):
                break

            if (A[p] < A[q + 1]):
                p += 1
            else:
                tmp = A[q + 1]
                for j in range(q, p - 1, -1):
                    A[j + 1] = A[j]
                A[p] = tmp
                p += 1
                q += 1

    def merge(self, A, p, q, r):
        """
        1 2 4 2 3 5
        p   q     r
        """
        a1 = A[p:q + 1].copy()
        a2 = A[q + 1:r + 1].copy()

        a1.append(math.inf)
        a2.append(math.inf)

        i1 = 0
        i2 = 0

        for i in range(p, r + 1):
            if (a1[i1] > a2[i2]):
                A[i] = a2[i2]
                i2 += 1
            else:
                A[i] = a1[i1]
                i1 += 1
